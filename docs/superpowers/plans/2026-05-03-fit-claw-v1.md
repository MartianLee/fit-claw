> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
>
> **Issue tracking:** Tasks are mirrored as beads (`fc-*`). Use `bd ready` / `bd show <id>` / `bd close <id>` for state. Do not use TodoWrite.

**Goal:** Build v1 — agent-callable workout & body-measurement logging backend with an SSR dashboard, deployed on Mac mini.

**Architecture:** Bun + Hono HTTP server with `bun:sqlite` for storage. Bearer-token auth. JSX SSR dashboard with htmx + Chart.js. Cloudflare Tunnel for public ingress (`/import/*`), Tailscale for management/dashboard. Cron via Bun setInterval.

**Tech Stack:** Bun, Hono, `bun:sqlite`, hand-rolled SQL migrations, `@hono/zod-validator`, Hono JSX, htmx, Chart.js, `cloudflared`, Tailscale, `launchd`, `rclone`.

## Conventions

- Every domain function has a unit test. HTTP handlers have at least one happy-path + one auth-failure integration test.
- Tests use `bun test`. DB tests open `:memory:` SQLite per test.
- Migrations are forward-only SQL files, applied in lexical order. An `applied_migrations(filename, applied_at)` table tracks state.
- All money/measurement values are real numbers in metric (kg, cm, %).
- Timestamps stored as ISO 8601 strings in UTC; conversion to user timezone is presentation-layer.
- All HTTP errors use `{error: {code, message}}` JSON shape, surfaced via a single error-handler middleware.
- Commit message convention: `feat:`, `fix:`, `chore:`, `test:`, `docs:`. One commit per task (or per coherent step inside a task).

```ts
import type { Database } from 'bun:sqlite'
import { readdirSync, readFileSync } from 'node:fs'
import { join } from 'node:path'

export type MigrationFile = { name: string; sql: string }

export function loadMigrationsFromDir(dir: string): MigrationFile[] {
  return readdirSync(dir)
    .filter((f) => f.endsWith('.sql'))
    .sort()
    .map((name) => ({ name, sql: readFileSync(join(dir, name), 'utf8') }))
}

export function runMigrations(db: Database, files: MigrationFile[]) {
  db.exec(`CREATE TABLE IF NOT EXISTS applied_migrations (
    name TEXT PRIMARY KEY,
    applied_at TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%fZ','now'))
  );`)

  const applied = new Set(
    (db.query('SELECT name FROM applied_migrations').all() as {name:string}[]).map(r => r.name)
  )

  for (const f of files) {
    if (applied.has(f.name)) continue
    db.transaction(() => {
      db.exec(f.sql)
      db.run('INSERT INTO applied_migrations(name) VALUES (?)', [f.name])
    })()
  }
}
```

let _db: Database | null = null

export function getDb(path: string): Database {
  if (_db) return _db
  mkdirSync(dirname(path), { recursive: true })
  const db = new Database(path)
  db.exec('PRAGMA journal_mode = WAL;')
  db.exec('PRAGMA foreign_keys = ON;')

  runMigrations(db, files)
  _db = db
  return db
}

```ts
const enc = new TextEncoder()

export async function hashToken(token: string): Promise<string> {
  const data = enc.encode(token)
  const buf = await crypto.subtle.digest('SHA-256', data)
  return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2,'0')).join('')
}

export async function verifyToken(token: string, hash: string): Promise<boolean> {
  return (await hashToken(token)) === hash
}
```

import type { Database } from 'bun:sqlite'
import { hashToken } from './tokens'

  interface ContextVariableMap {
    userId: number
  }
}

export function bearerAuth(opts: BearerOpts): MiddlewareHandler {
  return async (c: Context, next) => {
    const auth = c.req.header('authorization') ?? ''
    const m = /^Bearer\s+(.+)$/i.exec(auth)
    if (!m) return c.json({ error: { code: 'unauthorized', message: 'missing bearer' } }, 401)
    const h = await hashToken(m[1])
    const row = opts.db.query('SELECT user_id FROM api_tokens WHERE token_hash = ?').get(h) as {user_id:number}|null
    if (!row) return c.json({ error: { code: 'unauthorized', message: 'invalid token' } }, 401)
    opts.db.run('UPDATE api_tokens SET last_used_at = strftime(\'%Y-%m-%dT%H:%M:%fZ\',\'now\') WHERE token_hash = ?', [h])
    c.set('userId', row.user_id)
    await next()
  }
}

import type { Database } from 'bun:sqlite'

export type Exercise = { id: number; canonical_name: string; body_part: string; equipment: string; is_bodyweight: number }

export type FindResult =
  | { matched: Exercise }
  | { matched?: undefined; proposed: { canonical_name: string; body_part: string; equipment: string } }

export function findOrProposeExercise(db: Database, query: string): FindResult {
  const q = query.trim().toLowerCase()
  const direct = db.query('SELECT * FROM exercises WHERE LOWER(canonical_name) = ?').get(q) as Exercise|null
  if (direct) return { matched: direct }
  const aliasHit = db.query(
    'SELECT e.* FROM exercise_aliases a JOIN exercises e ON e.id = a.exercise_id WHERE a.alias = ? COLLATE NOCASE'
  ).get(query) as Exercise|null
  if (aliasHit) return { matched: aliasHit }
  return { proposed: { canonical_name: query.trim().toLowerCase(), body_part: 'unknown', equipment: 'unknown' } }
}

export function confirmNewExercise(
  db: Database,
  input: { canonical_name: string; body_part: string; equipment: string; is_bodyweight?: boolean; aliases?: string[] },
): Exercise {
  return db.transaction(() => {
    const r = db.run(
      'INSERT INTO exercises(canonical_name, body_part, equipment, is_bodyweight) VALUES (?,?,?,?)',
      [input.canonical_name, input.body_part, input.equipment, input.is_bodyweight ? 1 : 0],
    )
    const id = Number(r.lastInsertRowid)
    for (const a of input.aliases ?? []) {
      db.run('INSERT INTO exercise_aliases(exercise_id, alias) VALUES (?, ?) ON CONFLICT DO NOTHING', [id, a])
    }
    return db.query('SELECT * FROM exercises WHERE id = ?').get(id) as Exercise
  })()
}

import type { Database } from 'bun:sqlite'

export type SetInput = {
  weight_kg: number; reps: number
  rpe?: number; rir?: number; rest_sec?: number; tempo?: string; notes?: string
}

export type CreateEntryInput = {
  user_id: number
  exercise_id: number
  sets: SetInput[]
  session_id?: number
  started_at?: string
  ended_at?: string
  notes?: string
}

export type WorkoutSession = { id: number; user_id: number; started_at: string; ended_at: string|null; notes: string|null }
export type WorkoutEntry = { id: number; session_id: number; exercise_id: number; sequence: number }
export type WorkoutSet = { id: number; entry_id: number; set_number: number; weight_kg: number; reps: number; rpe: number|null; rir: number|null; rest_sec: number|null; tempo: string|null; notes: string|null }

export function createWorkoutEntry(db: Database, input: CreateEntryInput) {
  return db.transaction(() => {
    let session: WorkoutSession
    if (input.session_id) {
      const s = db.query('SELECT * FROM workout_sessions WHERE id = ?').get(input.session_id) as WorkoutSession|null
      if (!s) throw new Error('session not found')
      session = s
    } else {
      const startedAt = input.started_at ?? new Date().toISOString()
      const r = db.run(
        'INSERT INTO workout_sessions(user_id, started_at, ended_at, notes) VALUES (?,?,?,?)',
        [input.user_id, startedAt, input.ended_at ?? null, input.notes ?? null],
      )
      session = db.query('SELECT * FROM workout_sessions WHERE id = ?').get(Number(r.lastInsertRowid)) as WorkoutSession
    }
    const seq = (db.query('SELECT COALESCE(MAX(sequence),0)+1 AS n FROM workout_entries WHERE session_id = ?').get(session.id) as {n:number}).n
    const er = db.run('INSERT INTO workout_entries(session_id, exercise_id, sequence) VALUES (?,?,?)', [session.id, input.exercise_id, seq])
    const entryId = Number(er.lastInsertRowid)
    const inserted: WorkoutSet[] = []
    let setNo = 1
    for (const s of input.sets) {
      const sr = db.run(
        'INSERT INTO workout_sets(entry_id, set_number, weight_kg, reps, rpe, rir, rest_sec, tempo, notes) VALUES (?,?,?,?,?,?,?,?,?)',
        [entryId, setNo++, s.weight_kg, s.reps, s.rpe ?? null, s.rir ?? null, s.rest_sec ?? null, s.tempo ?? null, s.notes ?? null],
      )
      inserted.push(db.query('SELECT * FROM workout_sets WHERE id = ?').get(Number(sr.lastInsertRowid)) as WorkoutSet)
    }
    const entry = db.query('SELECT * FROM workout_entries WHERE id = ?').get(entryId) as WorkoutEntry
    return { session, entry, sets: inserted }
  })()
}

    if (q.exercise_id && entries.length === 0) continue
    const grouped = entries.map(e => ({
      entry: e,
      sets: db.query('SELECT * FROM workout_sets WHERE entry_id = ? ORDER BY set_number').all(e.id) as WorkoutSet[],
    }))
    out.push({ session: s, entries: grouped })
  }
  return out
}

export function recentWorkouts(db: Database, q: { user_id: number; days?: number }) {
  const days = q.days ?? 7
  const cutoff = new Date(Date.now() - days * 86400_000).toISOString()
  return queryWorkouts(db, { user_id: q.user_id, date_from: cutoff })
}

export function updateSet(db: Database, id: number, patch: Partial<SetInput>): WorkoutSet {
  const fields = ['weight_kg','reps','rpe','rir','rest_sec','tempo','notes'] as const
  const sets: string[] = []; const vals: any[] = []
  for (const f of fields) {
    if ((patch as any)[f] !== undefined) { sets.push(`${f} = ?`); vals.push((patch as any)[f]) }
  }
  if (sets.length === 0) return db.query('SELECT * FROM workout_sets WHERE id = ?').get(id) as WorkoutSet
  vals.push(id)
  db.run(`UPDATE workout_sets SET ${sets.join(', ')} WHERE id = ?`, vals)
  return db.query('SELECT * FROM workout_sets WHERE id = ?').get(id) as WorkoutSet
}

export function deleteSet(db: Database, id: number): void {
  db.run('DELETE FROM workout_sets WHERE id = ?', [id])
}

import type { Database } from 'bun:sqlite'

export function epley1Rm(weight_kg: number, reps: number): number {
  if (weight_kg <= 0 || reps <= 0) return 0
  return weight_kg * (1 + reps / 30)
}

export type OneRmPoint = { date: string; est_1rm: number }

export function oneRmDailySeries(
  db: Database,
  q: { user_id: number; exercise_id: number; date_from?: string; date_to?: string },
): OneRmPoint[] {
  const rows = db.query(
    `SELECT substr(s.started_at,1,10) AS date, ws.weight_kg AS w, ws.reps AS r
     FROM workout_sets ws
     JOIN workout_entries e ON e.id = ws.entry_id
     JOIN workout_sessions s ON s.id = e.session_id
     WHERE s.user_id = ? AND e.exercise_id = ?
       AND (? IS NULL OR s.started_at >= ?)
       AND (? IS NULL OR s.started_at <= ?)`
  ).all(q.user_id, q.exercise_id, q.date_from ?? null, q.date_from ?? null, q.date_to ?? null, q.date_to ?? null) as {date:string; w:number; r:number}[]
  const dailyMax = new Map<string, number>()
  for (const row of rows) {
    const v = epley1Rm(row.w, row.r)
    const cur = dailyMax.get(row.date) ?? 0
    if (v > cur) dailyMax.set(row.date, v)
  }
  return Array.from(dailyMax.entries()).sort(([a],[b]) => a.localeCompare(b)).map(([date, est_1rm]) => ({ date, est_1rm }))
}

    },

         LEFT JOIN workout_entries e ON e.session_id = s.id
         LEFT JOIN workout_sets ws ON ws.entry_id = e.id

         GROUP BY substr(s.started_at,1,10) ORDER BY date`

      ).all(c.get('userId'))
      return c.json(rows)
    },
  )

  return r

```ts
import { loadConfig } from '../config'
import { Database } from 'bun:sqlite'
import { join } from 'node:path'
import { mkdirSync, existsSync } from 'node:fs'
import { spawnSync } from 'node:child_process'

const cfg = loadConfig()
const dir = './data/backups'
mkdirSync(dir, { recursive: true })
const stamp = new Date().toISOString().replace(/[:.]/g,'-')
const out = join(dir, `fit-claw-${stamp}.db`)

if (!existsSync(cfg.databasePath)) { console.error('db missing'); process.exit(1) }

const src = new Database(cfg.databasePath, { readonly: true })
src.run(`VACUUM INTO '${out}'`)
src.close()

const gz = `${out}.gz`
const r1 = spawnSync('gzip', [out])
if (r1.status !== 0) { console.error('gzip failed'); process.exit(1) }

if (cfg.backup.rcloneRemote) {
  const r2 = spawnSync('rclone', ['copy', gz, cfg.backup.rcloneRemote, '--quiet'])
  if (r2.status !== 0) { console.error('rclone failed'); process.exit(1) }
  // retention
  const r3 = spawnSync('rclone', ['delete', cfg.backup.rcloneRemote, '--min-age', `${cfg.backup.retentionDays}d`, '--quiet'])
  if (r3.status !== 0) console.error('retention prune failed (non-fatal)')
}

console.log(`backup ok: ${gz}`)
```

export function tailnetOnly(): MiddlewareHandler {
  return async (c, next) => {
    const ip = c.req.header('x-forwarded-for')?.split(',')[0]?.trim() ?? ''

  }
}
