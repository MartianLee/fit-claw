```ts
export function todayInTz(tz: string, now: Date = new Date()): string {
  return new Intl.DateTimeFormat('en-CA', {
    timeZone: tz,
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
  }).format(now)
}

export function hhmmInTz(tz: string, now: Date = new Date()): string {
  const parts = new Intl.DateTimeFormat('en-GB', {
    timeZone: tz,
    hour: '2-digit',
    minute: '2-digit',
    hourCycle: 'h23',
  }).formatToParts(now)
  const hour = parts.find((p) => p.type === 'hour')?.value ?? '00'
  const minute = parts.find((p) => p.type === 'minute')?.value ?? '00'
  return `${hour}:${minute}`
}

export function addDaysIso(date: string, days: number): string {
  const [y, m, d] = date.split('-').map(Number)
  const dt = new Date(Date.UTC(y, m - 1, d))
  dt.setUTCDate(dt.getUTCDate() + days)
  const yy = dt.getUTCFullYear()
  const mm = String(dt.getUTCMonth() + 1).padStart(2, '0')
  const dd = String(dt.getUTCDate()).padStart(2, '0')
  return `${yy}-${mm}-${dd}`
}
```

```ts
import type { Database } from 'bun:sqlite'
import { HttpError } from '../lib/http'
import { todayInTz } from '../lib/time'

export type Checkin = {
  id: number
  user_id: number
  date: string
  sleep_hours: number | null
  condition_score: number | null
  notes: string | null
}

export type LogCheckinInput = {
  user_id: number
  tz: string
  now?: Date
  date?: string | null
  sleep_hours?: number | null
  condition_score?: number | null
  notes?: string | null
}

export function logCheckin(db: Database, input: LogCheckinInput): Checkin {
  const date = input.date ?? todayInTz(input.tz, input.now)

  return db.transaction(() => {
    const existing = db
      .query('SELECT * FROM daily_checkins WHERE user_id = ? AND date = ?')
      .get(input.user_id, date) as Checkin | null

    const next = {
      sleep_hours:
        'sleep_hours' in input ? input.sleep_hours ?? null : existing?.sleep_hours ?? null,
      condition_score:
        'condition_score' in input
          ? input.condition_score ?? null
          : existing?.condition_score ?? null,
      notes: existing?.notes ?? null,
    }

    if (existing) {
      db.run(
        'UPDATE daily_checkins SET sleep_hours = ?, condition_score = ?, notes = ? WHERE id = ?',
        [next.sleep_hours, next.condition_score, next.notes, existing.id],
      )
      return db
        .query('SELECT * FROM daily_checkins WHERE id = ?')
        .get(existing.id) as Checkin
    }

    const result = db.run(
      'INSERT INTO daily_checkins(user_id, date, sleep_hours, condition_score, notes) VALUES (?, ?, ?, ?, ?)',
      [input.user_id, date, next.sleep_hours, next.condition_score, next.notes],
    )
    return db
      .query('SELECT * FROM daily_checkins WHERE id = ?')
      .get(Number(result.lastInsertRowid)) as Checkin
  })()
}
```

```ts
const PREFIX_RE = /^\[\d{2}:\d{2}\] /

function appendNote(existing: string | null, text: string, hhmm: string): string {
  const line = `[${hhmm}] ${text}`
  if (!existing) return line
  const lines = existing.split('\n')
  const lastText = lines[lines.length - 1].replace(PREFIX_RE, '')
  if (lastText === text) return existing
  return `${existing}\n${line}`
}

export type Checkin = {
  id: number
  user_id: number
  date: string
  sleep_hours: number | null
  condition_score: number | null
  notes: string | null
}

  tz: string
  now?: Date

  sleep_hours?: number | null
  condition_score?: number | null
  notes?: string | null
```

```ts
export type QueryCheckinsInput = {
  user_id: number
  tz: string
  now?: Date
  date_from?: string
  date_to?: string
}

const MAX_RANGE_DAYS = 365
const DEFAULT_WINDOW_DAYS = 30

export function queryCheckins(db: Database, input: QueryCheckinsInput): Checkin[] {
  const today = todayInTz(input.tz, input.now)
  const dateTo = input.date_to ?? today
  const dateFrom = input.date_from ?? addDaysIso(dateTo, -DEFAULT_WINDOW_DAYS)

  const spanDays = Math.round(
    (Date.parse(`${dateTo}T00:00:00Z`) - Date.parse(`${dateFrom}T00:00:00Z`)) / 86_400_000,
  )
  if (spanDays > MAX_RANGE_DAYS) {
    throw new HttpError(400, 'invalid_input', `range exceeds ${MAX_RANGE_DAYS} days`)
  }

  return db
    .query(
      `SELECT * FROM daily_checkins
       WHERE user_id = ? AND date >= ? AND date <= ?
       ORDER BY date ASC`,
    )
    .all(input.user_id, dateFrom, dateTo) as Checkin[]
}
```

```ts
import type { Database } from 'bun:sqlite'
import { Hono } from 'hono'
import { z } from 'zod'
import { logCheckin, queryCheckins } from '../domain/checkins'
import { zv } from '../lib/validator'

const DATE_RE = /^\d{4}-\d{2}-\d{2}$/

const LogCheckinSchema = z.object({
  date: z.string().regex(DATE_RE).nullish(),
  sleep_hours: z.number().min(0).max(24).nullish(),
  condition_score: z.number().int().min(1).max(5).nullish(),
  notes: z.string().max(2000).nullish(),
})

const QueryCheckinsSchema = z.object({
  date_from: z.string().regex(DATE_RE).optional(),
  date_to: z.string().regex(DATE_RE).optional(),
})

function userTimezone(db: Database, userId: number): string {
  const row = db
    .query('SELECT timezone FROM users WHERE id = ?')
    .get(userId) as { timezone: string } | null
  return row?.timezone ?? 'UTC'
}

export function checkinsRoutes(db: Database) {
  const routes = new Hono()

  routes.post('/log_checkin', zv('json', LogCheckinSchema), (c) => {
    const userId = c.get('userId') as number
    const tz = userTimezone(db, userId)
    const body = c.req.valid('json')
    return c.json(logCheckin(db, { user_id: userId, tz, ...body }))
  })

  routes.post('/query_checkins', zv('json', QueryCheckinsSchema), (c) => {
    const userId = c.get('userId') as number
    const tz = userTimezone(db, userId)
    const body = c.req.valid('json')
    return c.json(queryCheckins(db, { user_id: userId, tz, ...body }))
  })

  return routes
}
```
